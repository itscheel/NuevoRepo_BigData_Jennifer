In [7]:
lst = []
for year in range(2017,2021):
    for autor in autores:
        publicaciones = dblp_api.search_publication('year:'+str(year)+': type:Journal_Articles: author:'+autor+':',max_search_results=1000)
        for result in range(len(publicaciones.results)):
            for j in range(len(publicaciones.results[result].publication.authors)):
                dic = {}
                dic['key'] = publicaciones.results[result].publication.key
                dic['title'] = publicaciones.results[result].publication.title
                dic['year'] = publicaciones.results[result].publication.year
                dic['author'] = publicaciones.results[result].publication.authors[j].name['text']
                lst.append(dic)

Generación de grafos¶

In [46]:
G = []
GN = []
communities = []
lst_df = []
for i in range(2017,2021):
    n = i-2017
    print(n)
    df = historico[historico['year']<=i]
    aux = pd.DataFrame(df['author'].value_counts()).reset_index()
    lst_autores = aux[aux['author']>=5]['index'].to_list()
    df = df[df['author'].isin(lst_autores)].reset_index(drop=True)
    aux = pd.DataFrame(df['key'].value_counts()).reset_index()
    lst_keys = aux[aux['key']>1]['index'].to_list()
    df = df[df['key'].isin(lst_keys)].reset_index(drop=True)
    
    coautorias = df[['key','author']].merge(df[['key','author']],on='key',how='left')
    coautorias = coautorias[coautorias['author_x'] != coautorias['author_y']].reset_index(drop=True)
    
    G.append(nx.Graph())
    data = [(x,y) for x,y in zip(coautorias['author_x'],coautorias['author_y'])]
    G[n].add_edges_from(data)
    
    GN.append(nxcom.girvan_newman(G[n]))
    communities.append(next(GN[n]))
    
    lst_df.append(df)
